   def print_Dirs(masterpath):
      global pathfiles
      files=os.listdir(masterpath)
      for x in files:
          if os.path.isdir(masterpath+'/'+x):
              print_Dirs(masterpath+'/'+x)
          else:
              pathfiles.append(masterpath+'//'+x)
